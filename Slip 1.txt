Q.1) Write an AngularJS script for addition of two numbers using ng-init, ng-model &
ng-bind. And also demonstrate ng-show, ng-disabled, ng-click directives on button
component. 



Index.html

<!DOCTYPE html>
<html ng-app>
<head>
    <title>Addition in AngularJS</title>
    <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>
</head>
<body ng-init="a=0; b=0; result=0; showResult=false">

    <h2>Addition of Two Numbers Using AngularJS</h2>

    <div>
        Enter Number 1:
        <input type="number" ng-model="a" />

        <br><br>

        Enter Number 2:
        <input type="number" ng-model="b" />

        <br><br>

        <button 
            ng-click="result = (a + b); showResult = true"
            ng-disabled="a == 0 && b == 0">
            Add Numbers
        </button>

        <br><br>

        <div ng-show="showResult">
            <strong>Result: </strong>
            <span ng-bind="result"></span>
        </div>
    </div>

</body>
</html>








Q.2) Create a Node.js application that reads data from multiple files asynchronously
using promises and async/await.



const fs = require('fs').promises;

// Function to read files
async function readMultipleFiles() {
    try {
        const file1 = await fs.readFile('file1.txt', 'utf-8');
        const file2 = await fs.readFile('file2.txt', 'utf-8');

        console.log("Contents of File 1:");
        console.log(file1);

        console.log("\nContents of File 2:");
        console.log(file2);

    } catch (error) {
        console.error("Error reading files:", error);
    }
}

// Function call
readMultipleFiles();
