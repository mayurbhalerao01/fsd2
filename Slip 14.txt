Q.1) Create Angular application that print the name of students who got 85% using filter
and map method. 




app.component.ts
import { Component } from '@angular/core';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html'
})
export class AppComponent {
  students = [
    { name: "Rohan", percentage: 85 },
    { name: "Amit", percentage: 78 },
    { name: "Sneha", percentage: 85 },
    { name: "Neha", percentage: 90 }
  ];

  topperStudents = this.students
    .filter(s => s.percentage === 85)
    .map(s => s.name);
}

app.component.html
<h3>Students who scored 85%:</h3>
<ul>
  <li *ngFor="let s of topperStudents">
    {{ s }}
  </li>
</ul>








Q.2) Develop an Express.js application that defines routes for Create, Update operations
on a resource (Employee).





const express = require('express');
const app = express();

app.use(express.json());

let employees = [];

// Create Employee
app.post('/employees', (req, res) => {
    const emp = req.body;
    employees.push(emp);
    res.send("Employee added");
});

// Update Employee
app.put('/employees/:id', (req, res) => {
    const empId = req.params.id;
    const updated = req.body;

    const index = employees.findIndex(e => e.id == empId);

    if (index !== -1) {
        employees[index] = updated;
        res.send("Employee updated");
    } else {
        res.send("Employee not found");
    }
});

app.listen(3000, () => {
    console.log("Server running on port 3000");
});
