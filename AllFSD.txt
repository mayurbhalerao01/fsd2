Q. 6 	) Find a company with a workforce greater than 30 in the array (use find by id method).
company.service.ts
import { Injectable } from '@angular/core'; import { Observable, of } from 'rxjs'; export interface Company {
id: number; name: string;
workforce: number;
}
@Injectable({ providedIn: 'root'
})
export class CompanyService { private companies: Company[] = [
{ id: 1, name: 'TechCorp', workforce: 25 },
{ id: 2, name: 'InnoSoft', workforce: 40 },
{ id: 3, name: 'WebSolutions', workforce: 15 },
{ id: 4, name: 'AlphaSystems', workforce: 50 },
{ id: 5, name: 'MegaTech', workforce: 60 }
 
];
constructor() { }
getCompanies(): Observable<Company[]> { return of(this.companies);
}
}
app.component.ts
import { Component, OnInit } from '@angular/core';
import { CompanyService, Company } from './services/company.service'; @Component({
selector: 'app-root',
templateUrl: './app.component.html', styleUrls: ['./app.component.css']
})
export class AppComponent implements OnInit { companies: Company[] = [];
filteredCompanies: Company[] = [];
constructor(private companyService: CompanyService) { } ngOnInit(): void { this.companyService.getCompanies().subscribe(data => {
this.companies = data;
});
}
findCompanies(): void {
 
this.filteredCompanies = this.companies.filter(company => company.workforce > 30);
}
}
app.component.html
<div style="text-align:center; margin:20px;">
<h1>Companies with Workforce Greater Than 30</h1>
<button (click)="findCompanies()">Show Companies</button>
<table *ngIf="filteredCompanies.length > 0" border="1" cellpadding="5" cellspacing="0" style="margin:auto; margin-top:20px; width:70%;">
<tr>
<th>ID</th>
<th>Name</th>
<th>Workforce</th>
</tr>
<tr *ngFor="let company of filteredCompanies">
<td>{{ company.id }}</td>
<td>{{ company.name }}</td>
<td>{{ company.workforce }}</td>
</tr>
</table>

No company found with workforce greater than 30.
